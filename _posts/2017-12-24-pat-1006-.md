1006. Sign In and Sign Out (25)

      这题没啥坑，写就行

```
#include<stdio.h>
#include<math.h>
#include<iostream>
#include<string.h>
#include<algorithm>
#include<queue>
#include<map>
using namespace std;
string str1,str2;
string pattern;

int main() 
{
  int n;
  cin>>n;
  if(n==0) return 0;
  string temp1,temp2;
  string ans1,ans2;
  cin>>pattern>>temp1>>temp2;
  ans1 = temp1, ans2 = temp2;
  str1 = str2 = pattern;
  n--;
  while(n--) {
    cin>>pattern>>temp1>>temp2;
    if(temp1<ans1) {
      ans1 = temp1;
      str1 = pattern;
    }
    if(temp2>ans2) {
      ans2 = temp2;
      str2 = pattern;
    }
  }
  cout<<str1<<" "<<str2<<endl;
}
```
