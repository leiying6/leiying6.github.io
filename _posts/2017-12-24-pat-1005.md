#include<stdio.h>
#include<math.h>
#include<iostream>
#include<string.h>
#include<algorithm>
#include<queue>
#include<map>
using namespace std;
string mp[15];

int main() 
{
  mp[0] = "zero";
  mp[1] = "one";
  mp[2] = "two";
  mp[3] = "three";
  mp[4] = "four";
  mp[5] = "five";
  mp[6] = "six";
  mp[7] = "seven";
  mp[8] = "eight";
  mp[9] = "nine";
  char number[105];
  scanf("%s",number);
  int sum = 0;
  for(int i=0;i<strlen(number);i++) {
    sum += number[i]-'0';
  }
  if(sum == 0) {
    cout<<mp[0]<<endl;
    return 0;
  }
  int temp = sum;
  int cnt = 0;
  int ans[15] = {0};
  while(temp) {
    int aa = temp%10;
    ans[cnt++] = aa;
    temp/=10;
  }
  for(int i=cnt-1;i>=0;i--) {
    if(i!=0) cout<<mp[ans[i]]<<' ';
    else cout<<mp[ans[i]]<<endl;
  }
}
